# --- Entradas ---
moneda = int(input("¿Qué moneda usará? (1 para Bs / 2 para $): "))
monto = int(input("¿Cuánto dinero quiere sacar?: "))
cuenta = int(input("Tipo de cuenta (1: Ahorro / 2: Corriente): "))

# --- Validaciones ---

if monto % 10 != 0:
    print("Error: Monto no compatible con denominaciones disponibles")
    exit()

if moneda == 1 and monto > 10000:
    print("Transacción denegada: Excede limite diario")
    exit()

if moneda == 2 and monto > 500:
    print("Transacción denegada: Excede limite diario")
    exit()

# --- Cuentas de los billetes y comisión ---

if cuenta == 2:
    comision = monto * 0.05
else:
    comision = 0

total_final = monto + comision

c100 = monto // 100
sobra = monto % 100

c50 = sobra // 50
sobra = sobra % 50

c20 = sobra // 20
sobra = sobra % 20

c10 = sobra // 10

# --- Salida de datos ---

match moneda:
    case 1:
        texto_moneda = "Bolívares"
        signo = "Bs"
    case 2:
        texto_moneda = "Dólares"
        signo = "$"

print("--------------------------------")
print(f"RECIBO DE RETIRO EN {texto_moneda}")
print("--------------------------------")
print(f"Billetes de 100: {c100}")
print(f"Billetes de 50: {c50}")
print(f"Billetes de 20: {c20}")
print(f"Billetes de 10: {c10}")
print("--------------------------------")
print(f"Total que se quitó de la cuenta: {total_final:.2f} {signo}")
print("--------------------------------")
